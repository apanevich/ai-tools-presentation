\section{Доступность из России}

\begin{frame}{Ограничения и обходные пути}
  \begin{itemize}
    \item \textbf{Геоблокировки} — ChatGPT, Gemini, Claude, часть API требуют VPN.
    \item \textbf{Оплата} — российские карты не принимаются; нужны зарубежные карты, криптовалюта или посредники.
    \item \textbf{Регистрация} — часто требуется зарубежный номер телефона.
    \item \textbf{Локальные альтернативы} — работают без VPN, но уступают в качестве.
    \item \textbf{OpenRouter / прокси-API} — агрегаторы, позволяют обходить блокировки.
  \end{itemize}
\end{frame}

\section{Примеры использования}

\begin{frame}{В учёбе}
  \begin{itemize}
    \item \textbf{Разбор сложных концепций}
    \item \textbf{Решение задач} — пошаговые решения по математике, физике, программированию.
    \item \textbf{Генерация учебных материалов} — конспекты, шпаргалки, тестовые вопросы для самопроверки.
    \item \textbf{Редактирование и проверка текстов} — курсовые, рефераты, отчёты: грамматика, стиль, логика.
  \end{itemize}
\end{frame}

\begin{frame}{В работе и научных проектах}
  \begin{itemize}
    \item \textbf{Советы по научной работе} — структура статей, выбор методологии, проверка гипотез.
    \item \textbf{Написание и редактирование текстов} — переформулировка, академический стиль, исправление ошибок.
    \item \textbf{Генерация идей и гипотез} — преодоление ментальных блоков, предложение направлений для исследований.
    \item \textbf{Форматирование и подготовка материалов} — оформление ссылок, таблиц, графиков.
  \end{itemize}
\end{frame}

\begin{frame}{Планирование и исследования}
  \begin{itemize}
    \item \textbf{Разбивка задач на шаги} — составление детального плана работы над проектом.
    \item \textbf{Оценка сроков} — прикидка времени на каждый этап, построение реалистичного графика.
    \item \textbf{Структурирование исследований} — логическая последовательность от постановки задачи до анализа результатов.
  \end{itemize}
  \vspace{0.5em}
  \small
  ИИ-инструменты поддерживают авторов на всех этапах — от идеи до готового текста.
\end{frame}

\section{Практические выводы}

\begin{frame}{Что получилось хорошо}
  \begin{itemize}
    \item Все чат-интерфейсы работают стабильно и дают качественные ответы.
    \item Связка \textbf{Continue + OpenRouter} — удобное переключение моделей прямо из VS Code.
    \item Российские решения \textbf{Koda} и \textbf{SourceCraft} работают без VPN.
    \item \textbf{Ollama} запускается без проблем, модели скачиваются и работают локально.
  \end{itemize}
\end{frame}

\begin{frame}{С чем были сложности}
  \begin{itemize}
    \item Некоторые зарубежные сервисы требуют VPN или зарубежную карту для оплаты.
    \item Локальные модели маленького размера (1.5B) заметно уступают по качеству большим облачным.
    \item API-доступ к моделям требует настройки и понимания тарифов и токенов.
    \item Галлюцинации и устаревшие данные — нужна обязательная верификация ответов.
  \end{itemize}
\end{frame}

\begin{frame}{Рекомендации аудитории}
  \begin{enumerate}
    \item Начните с одного чата (ChatGPT, DeepSeek или YandexGPT) и одного IDE-ассистента (Continue или Koda).
    \item Заведите личный набор «эталонных задач» для сравнения моделей.
    \item Для чувствительных данных используйте локальные модели (Ollama) или российские платформы.
    \item Всегда проверяйте код, факты и ссылки — ИИ может ошибаться.
    \item Используйте ИИ как помощника, а не замену собственному мышлению.
  \end{enumerate}
\end{frame}
